@namespace-стиль кестесінде қолданылатын XML аттар кеңістігін анықтайтын ережелер. Олар CSS таңдаушыларын белгілі бір аттар кеңістігіне жататын элементтермен шектеу үшін қолданылады. Namespace ' s негізінен SVG, MathML немесе XML-мен HTML5 сияқты көптеген атаулардан тұратын құжаттармен өзара әрекеттесу болған кезде пайдалы.

@rules and modifiers

@charset @import @namespace @media @supports @document @page @font-face @keyframes @viewport @counter-style @font-feature-values

Syntax

@namespace url('http://www.w3.org/1999/xhtml');
@namespace svg url('http://www.w3.org/2000/svg');

/* This matches all XHTML <a> elements, as XHTML is the default unprefixed namespace */
a {
}

/* This matches all SVG <a> elements */
svg|a {
}

/* This matches both XHTML and SVG <a> elements */
*|a {
}
