CSS-тегі @media ережесі операторлардың жиынтығын, CSS блогын жақшалармен шектелген, бұқаралық ақпарат құралдарының сұранысымен анықталған Шартпен байланыстырады. @media ережесін CSS-тің жоғарғы деңгейінде ғана емес, сонымен қатар шартты ережелер тобының кез-келген бөлігінде де қолдануға болады.

@rules and modifiers

@charset @import @namespace @media @supports @document @page @font-face @keyframes @viewport @counter-style @font-feature-values

Syntax

@media <media-query-list> {
    <group-rule-body>
}

где
<media-query-list> = <media-query>#

где
<media-query> = <media-condition> | [ not | only ]? <media-type> [ and <media-condition-without-or> ]?

где
<media-condition> = <media-not> | <media-and> | <media-or> | <media-in-parens>
<media-type> = <ident>
<media-condition-without-or> =  = not <media-in-parens>
<media-and> = <media-in-parens> [ and <media-in-parens> ]+
<media-or> = <media-in-parens> [ or <media-in-parens> ]+
<media-in-parens> = (  = ( [ <mf-plain> | <mf-boolean> | <mf-range> ] )
<general-enclosed> = [ <function-token> <any-value> ) ] | ( <ident> <any-value> )

где
<mf-plain> = <mf-name> :  = <mf-name>
<mf-range> = <mf-name> [ '<' | '>' ]? '='? ' '='? <mf-name> '>' '='? <mf-value>

где
<mf-name> = <ident>
<mf-value> = <number> | <dimension> | <ident> | <ratio>
